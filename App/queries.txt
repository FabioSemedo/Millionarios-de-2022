1.- Bilionarios self made ordenados por prim enroll e depois por tert enroll

select b.first_name, b.full_name, b.wealth, e.grs_prm_enroll, e.grs_trt_enroll from (Billionaires b join Cities c on b.cityID = c.cityID join EconomicDetails e on c.countryID = e.countryID) order by grs_prm_enroll desc, grs_trt_enroll desc

2.- Percentagem de bilionarios relativa Ã  populacao, ordenado alfabeticamente por pais

select o.name, (count(b.billionaireID) / e.population) as ratio from (Billionaires b join Cities c on b.cityID = c.cityID join on EconomicDetails e on b.countryID = e.countryID join Countries o on e.countryID = o.countryID) group by population order by o.name desc

3.- Bilionarios com idade maior do que a esperanca de vida e que cuja envolve trabalhos perigosos (exemplo: mineracao)

select b.industry, count(b.billionaireID) as quantity from (Billionaires b join Cities c on b.cityID = c.cityID join EconomicDetails e on c.countryID = e.countryID) where b.age > e.life_expect and (b.industry = "Metals & Mining" or b.industry = "Construction & Engineering") group by b.industry order by b.industry desc

5.- Total de dinheiro do bilionarios (pool) em funcao de industria

select industry, sum(wealth) as total from Billionaires group by industry order by industry

6.- Percentagem de bilionarios por continente

select o.continent, (count(b.billionaireID) / 2591 as percentage from (Billionaires b join Cities c on b.cityID = c.cityID join Countries o on c.countryID = o.countryID) group by o.continent order by o.continent

7.- Industrias mais populares (count) do midwest dos estados unidos, dos bilionarios ordenadas alfa cresc

select b.industry, count(b.billionaireID) as quantity from (Billionaires b join Cities c on b.cityID = c.cityID join USCities uc on c.cityID = uc.cityID join USStates s on uc.StateID = s.StateID) group by b.industry order by b.industry
